  CREATE TABLE Users (	  userId PRIMARY KEY, 	  username varchar(50) unique not null,	  firstName varchar(50) not null,	  lastName varchar(50) not null,	  email varchar(200) not null,	  userPassword varchar(50) not null,   );CREATE TABLE Post (	  postId varchar(50) PRIMARY KEY, 	  userOwner varchar(50) references Users(userId),	  postDate timestamp not null,	  title varchar(200) not null,	  postInfo varchar(1000) not null  );  CREATE TABLE React (	  reactId varchar(50) PRIMARY KEY, 	  userOwner varchar(50) references Users(userId),	  post varchar(50) references Post(postId)  );  CREATE TABLE Reply (	  replyId varchar(50) PRIMARY KEY, 	  userOwner varchar(50) references Users(userId),	  post varchar(50) references Post(postId),	  replyMessage varchar(500) not null  );
